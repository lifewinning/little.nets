I64 GrRGB24(I64 r,I64 g, I64 b)
{
  I64 i=r+g+b,k,c;
  if (!i)
    return BLACK;
  k=RandU16;
  i=(i*k)>>16;
  if (i<=r)
    c=RED;
  else {
    i-=r;
    if (i<=g)
      c=GREEN;
    else
      c=BLUE;
  }
  k=RandU16;
  i=r*r+g*g+b*b;
  if (i<k>>2)
    return BLACK;
  else if (i<k>>1)
    return c;
  else
    return c+8;
}
